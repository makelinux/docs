580int vvv()
581{
582 int a = 0;
583 visit([&a](auto&& arg) {assert(++a == 1);}, variant<int> {1});
584 assert(a == 1);
585 assert(max(0,1) == 1);
586 vector<var_t> vec = {10, 15l, 1.5, "hello"};
587 for (auto& v : vec) {
588
589 // 1. void visitor, only called for side-effects (here, for I/O)
590 visit([](auto&& arg) {cout << arg;}, v);
591
592 // 2. value-returning visitor, demonstrates the idiom of returning another variant
593 var_t w = visit([](auto&& arg) -> var_t {return arg + arg;}, v);
594
595 // 3. type-matching visitor: a lambda that handles each type differently
596 cout << ". After doubling, variant holds ";
597 visit([](auto&& arg) {
598 using T = decay_t<decltype(arg)>;
599 if constexpr(is_same_v<T, int>)
600 cout << "int with value " << arg << '\n';
601 else if constexpr(is_same_v<T, long>)
602 cout << "long with value " << arg << '\n';
603 else if constexpr(is_same_v<T, double>)
604 cout << "double with value " << arg << '\n';
605 else if constexpr(is_same_v<T, string>)
606 cout << "string with value " << arg << '\n';
607 else
608 static_assert(always_false_v<T>, "non-exhaustive visitor!");
609 }, w);
610 }
611
612 for (auto& v : vec) {
613 // 4. another type-matching visitor: a class with 3 overloaded operator()'s
614 visit(overloaded {
615 [](auto arg) { cout << arg << ' '; },
616 [](double arg) { cout << fixed << arg << ' '; },
617 [](const string & arg) { cout << arg << ' '; },
618 }, v);
619 }
620 return 0;
621}
