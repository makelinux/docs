507 () // optional list of arguments
508 {}; // body

510 // Primitive named lambdas are just like closure functions:
511 // https://en.wikipedia.org/wiki/Closure_(computer_programming)

513 // declaration like a function:
514 // void closure() { };
515 auto closure = [] {};
516
517 closure();

533
534 // annotated expanded empty inline lambda call:
535 [] // capture
536 () // optional list of arguments
537 -> void // optional return value
538 {} // body
539 (); // call with arguments
540
541 // annotated expanded sample inline lambda call:
542 c = // result
543 [c] // capture
544 (int a) // an argument

568 assert([=] { return i; }() == 2);
